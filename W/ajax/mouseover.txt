<h2>The XMLHttp</h2>
<button type="button" onmouseover="load()">Load Content</button>
</div>
<script>
function load(){
const xhttp = new XMLHttpRequest();
xhttp.onload=function(){
document.getElementById("demo").innerHTML=this.responseText;
}
xhttp.open("GET","ajax_info.txt",true);
xhttp.send();
}
</script>
