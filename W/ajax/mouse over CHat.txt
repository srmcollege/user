    <h2>The XMLHttpRequest Example</h2>
    <button type="button" onclick="loadContent()">Load Content on Click</button>
    <div id="demo">
        <h3>Mouse Over to Load Content</h3>
        <div onmouseover="loadContent()">
            Hover over this text to load content.
        </div>
    </div>
    <script>
        function loadContent() {
            const xhttp = new XMLHttpRequest();
            xhttp.onload = function() {
                document.getElementById("demo").innerHTML += "<h4>Content Loaded:</h4>" + this.responseText;
            }
            xhttp.open("GET", "ajax_info.txt", true);
            xhttp.send();
        }
    </script>
