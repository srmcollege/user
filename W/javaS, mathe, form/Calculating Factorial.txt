    <script>
        function factorial(n) {
            if (n < 0) return "Undefined for negative numbers";
            let result = 1;
            for (let i = 1; i <= n; i++) {
                result *= i;
            }
            return result;
        }
    </script>

    <p>Factorial of 5: <script>document.write(factorial(5));</script></p>
