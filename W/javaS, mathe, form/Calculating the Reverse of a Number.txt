    <script>
        function reverseNumber(num) {
            return parseInt(num.toString().split('').reverse().join('')) * Math.sign(num);
        }
    </script>

<body>
    <h1>Reverse of a Number</h1>
    <p>Reverse of 1234: <script>document.write(reverseNumber(1234));</script></p>
