DECLARE
   v_local NUMBER := 100; -- Local variable
BEGIN
   DBMS_OUTPUT.PUT_LINE('Local variable value: ' || v_local);
END;


------------------------


-- Package specification
CREATE OR REPLACE PACKAGE global_vars AS
   v_global NUMBER;  -- Global variable
END global_vars;
/

-- Package body
CREATE OR REPLACE PACKAGE BODY global_vars AS
   v_global NUMBER := 100;  -- Initialize global variable
END global_vars;
/

-- Using global variable in a block
BEGIN
   DBMS_OUTPUT.PUT_LINE('Global variable value: ' || global_vars.v_global);
END;
