
CREATE TABLE employees32 (employee_id NUMBER PRIMARY KEY,employee_name VARCHAR2(50),salary NUMBER);


CREATE OR REPLACE TRIGGER trg_before_update_salary
BEFORE UPDATE OF salary ON employees32
FOR EACH ROW
BEGIN
    DBMS_OUTPUT.PUT_LINE('Old Salary: ' || :OLD.salary); 
    DBMS_OUTPUT.PUT_LINE(' New Salary: ' || :NEW.salary);
    DBMS_OUTPUT.PUT_LINE(' Salary Difference: ' || (:NEW.salary - :OLD.salary));
END;
/

SET SERVEROUTPUT ON;

INSERT INTO employees32 (employee_id, employee_name, salary) VALUES (1, 'Alice', 400000);

UPDATE employees32 SET salary = 450000 WHERE employee_id = 1;
