CREATE TABLE employee (employee_id NUMBER PRIMARY KEY,employee_name VARCHAR2(100),employee_department VARCHAR2(100),employee_salary NUMBER
);

CREATE OR REPLACE PROCEDURE empupdate (eid IN NUMBER, ename IN VARCHAR2, edepartment IN VARCHAR2, esalary IN NUMBER) 
IS 
BEGIN 
    UPDATE employee22 
    SET 
        employee_name = ename, 
        employee_department = edepartment, 
        employee_salary = esalary
    WHERE employee_id = eid;
    
    IF SQL%ROWCOUNT = 0 THEN
        DBMS_OUTPUT.PUT_LINE('No employee found with ID: ' || eid);
    ELSE
        DBMS_OUTPUT.PUT_LINE('Employee updated successfully.');
    END IF;
END empupdate;
/

BEGIN 
    empupdate(11, 'Sanchit Kumar', 'Web Development', 25000);
END;
