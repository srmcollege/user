CREATE TABLE employee22 (
    employee_id NUMBER PRIMARY KEY,
    employee_name VARCHAR2(100),
    employee_department VARCHAR2(100),
    employee_salary NUMBER
);

CREATE OR REPLACE PROCEDURE empdelete (eid IN NUMBER) 
IS 
BEGIN 
    DELETE FROM employee22 
    WHERE employee_id = eid;
    
    IF SQL%ROWCOUNT = 0 THEN
        DBMS_OUTPUT.PUT_LINE('No employee found with ID: ' || eid);
    ELSE
        DBMS_OUTPUT.PUT_LINE('Employee deleted successfully.');
    END IF;
END empdelete;
/


BEGIN 
    empdelete(11);  
END;
/
 select *from employee22
