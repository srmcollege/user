CREATE TABLE employee22 (employee_id NUMBER PRIMARY KEY,employee_name VARCHAR2(100),employee_department VARCHAR2(100),employee_salary NUMBER);

CREATE OR REPLACE PROCEDURE empinsert (eid IN NUMBER, ename IN VARCHAR2, edepartment IN VARCHAR2, esalary IN NUMBER) 
IS 
BEGIN 
    INSERT INTO employee22 (employee_id, employee_name, employee_department, employee_salary) 
    VALUES (eid, ename, edepartment, esalary);
END empinsert;
/

BEGIN 
    empinsert(11, 'Sanchit', 'Web Development', 20000);
    DBMS_OUTPUT.PUT_LINE('Employee inserted successfully.');
END;
/


select *from employee22
