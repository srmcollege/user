DECLARE
    num NUMBER := 5;
    result NUMBER;

    FUNCTION factorial(n IN NUMBER) RETURN NUMBER IS
    BEGIN
        IF n = 0 THEN
            RETURN 1;
        ELSE
            RETURN n * factorial(n - 1);
        END IF;
    END factorial;

BEGIN
    result := factorial(num);
    DBMS_OUTPUT.PUT_LINE('The factorial of ' || num || ' is: ' || result);
END;
/
