DECLARE
   grade CHAR(1) := 'B';
BEGIN
   CASE grade
      WHEN 'A' THEN
         DBMS_OUTPUT.PUT_LINE('Excellent');
      WHEN 'B' THEN
         DBMS_OUTPUT.PUT_LINE('Good');
      WHEN 'C' THEN
         DBMS_OUTPUT.PUT_LINE('Fair');
      WHEN 'D' THEN
         DBMS_OUTPUT.PUT_LINE('Poor');
      ELSE
         DBMS_OUTPUT.PUT_LINE('Fail');
   END CASE;
END;
/
