-- Sample Table Creation
CREATE TABLE departments (
    dept_id NUMBER PRIMARY KEY,
    dept_name VARCHAR2(50)
);

-- Insert Sample Data
INSERT INTO departments (dept_id, dept_name) VALUES (1, 'HR');
INSERT INTO departments (dept_id, dept_name) VALUES (2, 'Finance');
INSERT INTO departments (dept_id, dept_name) VALUES (3, 'IT');
COMMIT;

-- PL/SQL Block Demonstrating Predefined Exception
DECLARE
    v_dept_name departments.dept_name%TYPE;
BEGIN
    SELECT dept_name INTO v_dept_name FROM departments; -- This will raise TOO_MANY_ROWS if multiple rows exist
    DBMS_OUTPUT.PUT_LINE('Department Name: ' || v_dept_name);
EXCEPTION
    WHEN TOO_MANY_ROWS THEN
        DBMS_OUTPUT.PUT_LINE('Error: More than one department found.');
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('An unexpected error occurred: ' || SQLERRM);
END;
/
