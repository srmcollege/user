-- Sample Table Creation
CREATE TABLE employees (
    emp_id NUMBER PRIMARY KEY,
    emp_name VARCHAR2(50),
    emp_salary NUMBER
);

-- Insert Sample Data
INSERT INTO employees (emp_id, emp_name, emp_salary) VALUES (1, 'Alice', 50000);
INSERT INTO employees (emp_id, emp_name, emp_salary) VALUES (2, 'Bob', 60000);
COMMIT;

-- PL/SQL Block Demonstrating User-Defined Exception
DECLARE
    high_salary_exception EXCEPTION;
    v_salary employees.emp_salary%TYPE;
BEGIN
    SELECT emp_salary INTO v_salary FROM employees WHERE emp_id = 1;

    IF v_salary > 55000 THEN
        RAISE high_salary_exception;
    END IF;

    DBMS_OUTPUT.PUT_LINE('Salary is acceptable: ' || v_salary);
EXCEPTION
    WHEN high_salary_exception THEN
        DBMS_OUTPUT.PUT_LINE('Error: Salary exceeds acceptable limits.');
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('An unexpected error occurred: ' || SQLERRM);
END;
/
