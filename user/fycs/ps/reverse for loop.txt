--reverse for loops##

1)
declare
var1 number;
begin 
var1 :=10;
for var2 in reverse 1..10
loop
dbms_output.put_line(var1*var2);
end loop;
end;

2)
declare
var1 number;
begin
var1:= 10;
for var2 in reverse 1..10
loop
dbms_output.put_line(var1*var2);
end loop;
end;
