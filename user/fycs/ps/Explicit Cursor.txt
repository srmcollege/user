DECLARE
	c_id Customer.ID%TYPE;
	c_name Customer.NAME%TYPE;
	c_salary Customer.SALARY%TYPE;
	
	CURSOR c_Customer IS SELECT ID,NAME,SALARY FROM Customer;
	
BEGIN
	OPEN c_Customer ;
		LOOP
			FETCH c_Customer INTO c_id ,c_name ,c_salary ;
			EXIT WHEN c_Customer %NOTFOUND;
			DBMS_OUTPUT.PUT_LINE(c_id ||'   '||c_name ||'   '||c_salary );
		END LOOP;
	CLOSE c_Customer ;             
END;

DECLARE
	CURSOR emp_cursor(P_DEPARTMENT_ID NUMBER)IS
	SELECT ID,NAME FROM employee WHERE ID=P_DEPARTMENT_ID;
	
	V_ID employee.ID%TYPE;
	V_NAME employee.NAME%TYPE;
BEGIN
	OPEN emp_cursor(1003);
		LOOP
			FETCH emp_cursor INTO V_ID,V_NAME;
			EXIT WHEN emp_cursor%NOTFOUND;
			DBMS_OUTPUT.PUT_LINE('Employee ID : '||V_ID||' NAME :'||V_NAME);
		END LOOP;
	CLOSE emp_cursor;
END;
