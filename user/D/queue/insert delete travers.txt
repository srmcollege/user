import time

queue = []
s = int(input("Enter number of elements: "))

for i in range(s):
    item = int(input(f"Enter Element {i + 1}: "))
    queue.append(item)

print("Queue is", queue)
print("Calculating size...")
time.sleep(1)
print("Size of queue is", len(queue))
time.sleep(1)

print("Dequeuing the element...")
if queue:
    print("Dequeued:", queue.pop(0))
else:
    print("Queue is empty")
time.sleep(0.5)

print("After Dequeue:", queue)
print("Size of queue after dequeue:", len(queue))

print("Traversing the queue:")
for i in range(len(queue)):
    print(f"Element {i + 1}: {queue[i]}")

if queue:
    print("Front End is", queue[0])
    print("Rear End is", queue[-1])
else:
    print("Queue is empty")
