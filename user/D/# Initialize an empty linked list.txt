# Initialize an empty linked list
linked_list = []

# Insert elements at the end
print("Insert")
list_size = int(input("Enter the size of the list: "))
for i in range(list_size):
    element = input("Enter the element: ")
    linked_list.append(element)  # Insert at the end

# Display the linked list
print("Display")
if not linked_list:
    print("Linked List is empty")
else:
    print("Current Linked List:", linked_list)

# Search for an element
print("Search")
search_element = input("Enter the element to search: ")
if search_element in linked_list:
    index = linked_list.index(search_element)
    print(f"Element '{search_element}' found in the linked list at index {index}")
else:
    print(f"Element '{search_element}' not found in the linked list")

# Delete the first occurrence of an element
print("Deletion")
remove_element = input("Enter the element you want to delete: ")
if remove_element in linked_list:
    linked_list.remove(remove_element)  # Remove the first occurrence
    print(f"Removed element: {remove_element}")
else:
    print("Element not found, nothing to delete.")

# Display the linked list after deletion
print("Display after Deletion")
if not linked_list:
    print("Linked List is empty")
else:
    print("Current Linked List after Deletion:", linked_list)
