import time

stack = []
n = int(input("Enter number of elements you want to enter: "))

for i in range(n):
    item = int(input(f"Enter Element {i + 1}: "))
    stack.append(item)

print("Stack is", stack)
print("Calculating size...")
time.sleep(1)
print("Size of stack is", len(stack))
time.sleep(1)
print()

print("Popping the element...")
if stack:
    print("Popped:", stack.pop())
else:
    print("Stack is empty")
time.sleep(0.5)
print()

print("After Pop:", stack)
print("Size of stack after pop:", len(stack))
print("Traversing the stack:")
for i in range(len(stack)):
    print(f"Element {i + 1}: {stack[i]}")
    
if stack:
    print("Top of the stack is", stack[-1])
else:
    print("Stack is empty")
