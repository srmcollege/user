# Initialize an empty doubly linked list
doubly_linked_list = []

# Insert elements at the end
print("Insert")
list_size = int(input("Enter the size of the list: "))
for i in range(list_size):
    element = input("Enter the element: ")
    doubly_linked_list.append(element)  # Insert at the end

# Display the doubly linked list
print("Display")
if not doubly_linked_list:
    print("Doubly Linked List is empty")
else:
    print("Current Doubly Linked List:", doubly_linked_list)

# Search for an element
print("Search")
search_element = input("Enter the element to search: ")
if search_element in doubly_linked_list:
    index = doubly_linked_list.index(search_element)
    print(f"Element '{search_element}' found in the doubly linked list at index {index}")
else:
    print(f"Element '{search_element}' not found in the doubly linked list")

# Delete the first occurrence of an element
print("Deletion")
remove_element = input("Enter the element you want to delete: ")
if remove_element in doubly_linked_list:
    doubly_linked_list.remove(remove_element)  # Remove the first occurrence
    print(f"Removed element: {remove_element}")
else:
    print("Element not found, nothing to delete.")

# Display the doubly linked list after deletion
print("Display after Deletion")
if not doubly_linked_list:
    print("Doubly Linked List is empty")
else:
    print("Current Doubly Linked List after Deletion:", doubly_linked_list)
