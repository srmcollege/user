package pract_6;

import java.io.FileInputStream;
import java.io.FileOutputStream;
import java.io.IOException;
import jxl.Sheet;
import jxl.Workbook;
import jxl.read.biff.BiffException;
import jxl.write.Label;
import jxl.write.WritableSheet;
import jxl.write.WritableWorkbook;
import jxl.write.WriteException;

public class Pract_6 {

    public static void main(String[] args) {
        try {
            FileInputStream is = new FileInputStream("D:\\TYCS\\STQA\\examdata1.xls");
            Workbook w = Workbook.getWorkbook(is);
            Sheet s = w.getSheet(0);
            String[][] a = new String[s.getRows()][s.getColumns()];

            FileOutputStream os = new FileOutputStream("D:\\TYCS\\STQA\\result2.xls");
            WritableWorkbook wwb = Workbook.createWorkbook(os);
            WritableSheet ws = wwb.createSheet("RESULT", 0);

            int c = 0;

            for (int i = 0; i < s.getRows(); i++) {
                for (int j = 0; j < s.getColumns(); j++) {
                    if (i >= 1) {
                        String b = s.getCell(3, i).getContents();
                        int x = Integer.parseInt(b);
                        if (x >= 60) {
                            c++;
                            break;
                        }
                    }
                    a[i][j] = s.getCell(j, i).getContents();
                    Label l2 = new Label(j, i - c, a[i][j]);
                    ws.addCell(l2);
                }
            }

            wwb.write();
            wwb.close();
            System.out.println("Student count: " + c);

        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}
